package Day7Lab1;

import java.util.Scanner;

class Date {
    private int date, month, year;
    Scanner sc = new Scanner(System.in);

    Date() {
        date = 1;
        month = 1;
        year = 2000;
        //AcceptInfo();
    }

    Date(int date, int month, int year) {
        this.date = date;
        this.month = month;
        this.year = year;
    }

    void AcceptInfo() {
        do {
            if ((!((date <= 31 && date >= 1) && (month <= 12 && month >= 1) && (year <= 3000 && year >= 1))))
                System.out.println("Entered the Wrong date \n Please Enter again");
            else
                System.out.print("Enter date in the format date month year = ");

            date = sc.nextInt();
            month = sc.nextInt();
            year = sc.nextInt();
        } while (!((date <= 31 && date >= 1) && (month <= 12 && month >= 1) && (year <= 3000 && year >= 1)));
    }

    void Display() {
        System.out.println("Date = " + date + " " + month + " " + year);
    }

}
